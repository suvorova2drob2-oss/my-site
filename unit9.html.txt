<script>
    // Массив: синоним из 1 задания + утверждение из 2 задания
    const allTasks = [
        { syn: "It is not a matter of worry for me at all", state: "the feeling of unease that AI induces...", id: "1", isA: true },
        { syn: "Positive mindset paying dividends", state: "the dividends that maintaining a positive mindset pays...", isA: false },
        { syn: "Uncertainty regarding the final result", state: "the unpredictability of where it will all end up", id: "2", isA: true },
        { syn: "As for me / from my perspective", state: "the consequences of collaboration...", id: "3", isA: true },
        { syn: "Participating in a joint shared task", state: "the naivety of artists...", isA: false },
        { syn: "Are fundamentally and closely linked", state: "how they go hand in hand", id: "5", isA: true }
    ];

    let step = 0;
    let isConfirmed = false;

    function update() {
        if(step < allTasks.length) {
            document.getElementById('synonym-text').innerText = "Find synonym: " + allTasks[step].syn;
            document.getElementById('statement-text').innerText = allTasks[step].state;
            document.getElementById('task-ui').classList.remove('active');
            isConfirmed = false;
        } else {
            document.getElementById('task-ui').innerHTML = "👑 SECTION A DONE!";
        }
    }

    function setReady() {
        if (!allTasks[step].isA) {
            alert("This belongs to another person! Click 'Doesn't match'.");
            return;
        }
        isConfirmed = true;
        document.getElementById('task-ui').classList.add('active');
    }

    function handleNoMatch() {
        if (allTasks[step].isA) {
            alert("This IS in the text! Click Match and find it.");
        } else {
            step++;
            update();
        }
    }

    document.querySelectorAll('.clickable').forEach(el => {
        el.addEventListener('click', function() {
            if (!isConfirmed) {
                alert("Click the statement or 'Matches' first!");
                return;
            }
            if(this.dataset.id === allTasks[step].id) {
                this.classList.add('found');
                step++;
                update();
            }
        });
    });

    update();
</script>
